#include <iostream>
using namespace std;

int Factorial(int n)
{
    int i;
    int factorial;

    factorial = 1;

    for(i = 1; i <= n; i++)
    {
        factorial = factorial * i;
    }

    return factorial;
}

int main()
{
    int n;
    int resultado;

    cout << "Ingrese un numero: ";
    cin >> n;

    resultado = Factorial(n);

------------------------------------------------------------
CÓDIGO C++

#include <iostream>
#include <string>
using namespace std;

int main()
{
    string cadena;
    int i, j;
    int n;
    char aux;

    cout << "Ingrese una palabra: ";
    cin >> cadena;

    n = cadena.length();

    for(i = 0; i < n - 1; i++)
    {
        for(j = 0; j < n - 1 - i; j++)
        {
            if(cadena[j] > cadena[j + 1])
            {
                aux = cadena[j];
                cadena[j] = cadena[j + 1];
                cadena[j + 1] = aux;
            }
        }
    }

    cout << "Cadena ordenada: " << cadena << endl;

    return 0;
}

Para i <- 1 Hasta n - 1 Hacer

    clave <- vector[i]

------------------------------------------------------------
CÓDIGO C++
------------------------------------------------------------

#include <iostream>
using namespace std;

int main()
{
    int n;
    int i, j;
    int clave;
    int vector[100];

    cout << "Ingrese la cantidad de elementos: ";
    cin >> n;

    for(i = 0; i < n; i++)
    {
        cout << "Elemento " << i + 1 << ": ";
        cin >> vector[i];
    }

    for(i = 1; i < n; i++)
    {
        clave = vector[i];
        j = i - 1;

        while(j >= 0 && vector[j] > clave)
        {
            vector[j + 1] = vector[j];
            j--;
        }

        vector[j + 1] = clave;
    }

    cout << "\nVector ordenado:\n";

    for(i = 0; i < n; i++)
    {
        cout << vector[i] << " ";
    }

    cout << endl;

    return 0;
}

#include <iostream>

using namespace std;

// Funcion que intercambia dos numeros
void INTERCAMBIAR(int &numero1, int &numero2)
{
    // Declaracion de variables
    int auxiliar;

    // Guardar el valor del primer numero
    auxiliar = numero1;

    // Copiar el segundo numero al primero
    numero1 = numero2;

    // Colocar el valor guardado en el segundo numero
    numero2 = auxiliar;
}

int main()
{
    // Declaracion de variables
    int numero1;
    int numero2;

    // Solicitar datos
    cout << "Ingrese el primer numero: ";
    cin >> numero1;

    cout << "Ingrese el segundo numero: ";
    cin >> numero2;

    // Mostrar los valores antes del intercambio
    cout << "\nAntes del intercambio" << endl;
    cout << "Numero 1: " << numero1 << endl;
    cout << "Numero 2: " << numero2 << endl;

    // Llamar a la funcion
    INTERCAMBIAR(numero1, numero2);

    // Mostrar los valores despues del intercambio
    cout << "\nDespues del intercambio" << endl;
    cout << "Numero 1: " << numero1 << endl;

#include <iostream>
#include <fstream>
#include <string>

using namespace std;

int main()
{
    ifstream archivo;

    string nombre;
    float nota;
    float suma;
    float promedio;
    int cantidad;

    suma = 0;
    cantidad = 0;

    archivo.open("estudiantes.txt");

    if(!archivo)
    {
        cout << "No se pudo abrir el archivo." << endl;
        return 1;
    }

    while(archivo >> nombre >> nota)
    {
        suma = suma + nota;
        cantidad++;
    }

    archivo.close();

    promedio = suma / cantidad;

    cout << "Promedio general: " << promedio << endl;

    return 0;
}

#include <iostream>
using namespace std;

void QuickSort(int vector[], int inicio, int fin)
{
    int i;
    int j;
    int pivote;
    int auxiliar;

    i = inicio;
    j = fin;
    pivote = vector[(inicio + fin) / 2];

    while(i <= j)
    {
        while(vector[i] < pivote)
        {
            i++;
        }

        while(vector[j] > pivote)
        {
            j--;
        }

        if(i <= j)
        {
            auxiliar = vector[i];
            vector[i] = vector[j];
            vector[j] = auxiliar;

            i++;
            j--;
        }
    }

    if(inicio < j)
    {
        QuickSort(vector, inicio, j);
    }

    if(i < fin)
    {
        QuickSort(vector, i, fin);
    }
}

#include <iostream>
using namespace std;

void BubbleSort(int vector[], int n)
{
    int i, j, auxiliar;

    for(i = 0; i < n - 1; i++)
    {
        for(j = 0; j < n - 1 - i; j++)
        {
            if(vector[j] > vector[j + 1])
            {
                auxiliar = vector[j];
                vector[j] = vector[j + 1];
                vector[j + 1] = auxiliar;
            }
        }
    }
}

void Merge(int vector[], int izquierda, int medio, int derecha)
{
    int i, j, k;

    int n1 = medio - izquierda + 1;
    int n2 = derecha - medio;

    int izquierdaVector[100];
    int derechaVector[100];

    for(i = 0; i < n1; i++)
    {
        izquierdaVector[i] = vector[izquierda + i];
    }

    for(j = 0; j < n2; j++)
    {
        derechaVector[j] = vector[medio + 1 + j];
    }

    i = 0;
    j = 0;
    k = izquierda;

    while(i < n1 && j < n2)
    {
        if(izquierdaVector[i] <= derechaVector[j])
        {
            vector[k] = izquierdaVector[i];
            i++;
        }
        else
        {
            vector[k] = derechaVector[j];
            j++;
        }

        k++;
    }

    while(i < n1)
    {
        vector[k] = izquierdaVector[i];
        i++;
        k++;
    }

    while(j < n2)
    {
        vector[k] = derechaVector[j];
        j++;
        k++;
    }
}

void MergeSort(int vector[], int izquierda, int derecha)
{
    int medio;

    if(izquierda < derecha)
    {
        medio = (izquierda + derecha) / 2;

        MergeSort(vector, izquierda, medio);

        MergeSort(vector, medio + 1, derecha);

        Merge(vector, izquierda, medio, derecha);
    }
}

    while(archivo >> nombre[cantidad]
                  >> nota1[cantidad]
                  >> nota2[cantidad]
                  >> nota3[cantidad])
    {
        promedio[cantidad] =
        (nota1[cantidad] +
         nota2[cantidad] +
         nota3[cantidad]) / 3.0;

        cantidad++;
    }

    archivo.close();

    for(i = 0; i < cantidad - 1; i++)
    {
        for(j = 0; j < cantidad - 1 - i; j++)
        {
            if(promedio[j] < promedio[j + 1])
            {
                auxPromedio = promedio[j];
                promedio[j] = promedio[j + 1];
                promedio[j + 1] = auxPromedio;

                auxNombre = nombre[j];
                nombre[j] = nombre[j + 1];
                nombre[j + 1] = auxNombre;
            }
        }
    }

    for(i = 0; i < cantidad; i++)
    {
        cout << i + 1 << ". "
             << nombre[i]
             << " Promedio: "
             << promedio[i]
             << endl;
    }

    for(i = 0; i < cantidad / 3; i++)
    {
        salida << nombre[i]
               << " "
               << promedio[i]
               << endl;
    }

    salida.close();

    return 0;
}

#include <iostream>
using namespace std;

void Ordenar(int vector[], int n)
{
    int i;
    int j;
    int auxiliar;

    for(i = 0; i < n - 1; i++)
    {
        for(j = 0; j < n - 1 - i; j++)
        {
            if((vector[j] % 2 != 0) && (vector[j + 1] % 2 == 0))
            {
                auxiliar = vector[j];
                vector[j] = vector[j + 1];
                vector[j + 1] = auxiliar;
            }
            else
            {
                if((vector[j] % 2 == 0) && (vector[j + 1] % 2 == 0))
                {
                    if(vector[j] > vector[j + 1])
                    {
                        auxiliar = vector[j];
                        vector[j] = vector[j + 1];
                        vector[j + 1] = auxiliar;
                    }
                }

                if((vector[j] % 2 != 0) && (vector[j + 1] % 2 != 0))
                {
                    if(vector[j] < vector[j + 1])
                    {
                        auxiliar = vector[j];
                        vector[j] = vector[j + 1];
                        vector[j + 1] = auxiliar;
                    }
                }
            }
        }
    }
}

int main()
{
    int n;
    int i;
    int vector[100];

    cout << "Ingrese la cantidad de elementos: ";
    cin >> n;

    for(i = 0; i < n; i++)
    {
        cout << "Elemento " << i + 1 << ": ";
        cin >> vector[i];
    }

    Ordenar(vector, n);

    cout << "\nVector ordenado:\n";

    for(i = 0; i < n; i++)
    {
        cout << vector[i] << " ";
    }

    cout << endl;

    return 0;
}
